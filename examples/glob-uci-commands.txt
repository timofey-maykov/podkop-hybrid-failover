# Пример: секция glob: AmneziaWG awg0 + резерв VLESS
# Подставьте свой vless:// из приложения (UUID, pbk, sid, sni).

uci set hybrid-failover.glob.failover_vpn_enabled='1'
uci del hybrid-failover.glob.failover_proxy_links 2>/dev/null
uci add_list hybrid-failover.glob.failover_proxy_links='vless://<UUID>@<HOST>:<PORT>?encryption=none&flow=xtls-rprx-vision&security=reality&sni=<SNI>&fp=chrome&pbk=<PUBLIC_KEY>&sid=<SHORT_ID>&type=tcp&headerType=none'

uci set hybrid-failover.glob.urltest_interrupt_exist_connections='1'
uci set hybrid-failover.glob.urltest_check_interval='20s'

uci commit hybrid-failover
/etc/init.d/hybrid-failover restart
